// REVISED VALIDATION LOGIC FOR OBSIDIAN ELITE document.getElementById('tradeForm').onsubmit = (e) => { e.preventDefault(); const editId = document.getElementById('editId').value; const ticker = document.querySelector('#tickerSelect .active')?.dataset.ticker; const type = document.querySelector('#typeSelect .active')?.dataset.type; const entryVal = document.getElementById('entry').value; const exitVal = document.getElementById('exit').value; // Helper to check if a toggle group has an active selection const getV = (id) => document.querySelector(`.toggle-group[data-id="${id}"] .active`)?.dataset.val; // 1. Mandatory Entry Validation if (!ticker || !type || !entryVal) { alert("STRICT GOVERNANCE: Ticker, Type, and Entry Price are mandatory to open a trade."); return; } // 2. Mandatory Exit Validation (Only triggers when trying to close an existing trade) if (editId !== "" && exitVal) { const setup = document.getElementById('setup').value; const contexts = ['vwap', 'ema5', 'mtf', 'pmRange', 'pdRange', 'pdc']; const missing = contexts.filter(id => !getV(id)); if (!setup || setup === "Select..." || missing.length > 0) { alert(`MANDATORY FIELDS REQUIRED FOR EXIT: \n- Setup must be selected\n- All technical context toggles must be active (${missing.join(', ').toUpperCase()})`); return; } } const tradeData = { stock: ticker, option: type, count: document.getElementById('count').value, entry: parseFloat(entryVal), setup: document.getElementById('setup').value, vwap: getV('vwap'), ema5: getV('ema5'), mtf: getV('mtf'), pmRange: getV('pmRange'), pdRange: getV('pdRange'), pdc: getV('pdc'), date: new Date().toISOString().split('T')[0] }; // Update or Create logic remains the same... if (editId !== "") { const t = trades[editId]; trades[editId] = { ...t, ...tradeData }; if (exitVal) { const ex = parseFloat(exitVal); trades[editId].exit = ex; trades[editId].status = 'CLOSED'; trades[editId].exitTime = t.exitTime || new Date().toLocaleTimeString('en-US', { hour12: false }); trades[editId].pnl = ((ex - tradeData.entry) * 100 * tradeData.count).toFixed(2); } } else { trades.unshift({ ...tradeData, status: 'OPEN', id: Date.now(), entryTime: new Date().toLocaleTimeString('en-US', { hour12: false }), pnl: 0 }); } resetForm(); render(); };

// --- UPDATED SUBMIT LOGIC --- document.getElementById('tradeForm').onsubmit = (e) => { e.preventDefault(); const editId = document.getElementById('editId').value; const getVal = (id) => { const el = document.querySelector(`[data-id="${id}"] .active`); return el ? el.dataset.val : null; }; // 1. OPENING NEW TRADE: Require Ticker, Type, and Entry if (!editId) { const ticker = document.querySelector('#tickerGrid .active')?.dataset.val; const type = getVal('optionType'); const entry = document.getElementById('entry').value; if (!ticker || !type || !entry) { alert("STRICT OPEN: Select Ticker, Type, and Entry Price!"); return; } trades.unshift({ id: Date.now(), date: getNYDate(), status: 'OPEN', stock: ticker, option: type, count: document.getElementById('count').value, entry: parseFloat(entry), entryTime: getNYTime(), exit: null, exitTime: null, pnl: 0, pnlPct: 0, setup: getVal('setup'), vwap: getVal('vwap'), ema512: getVal('ema512'), mtf: getVal('mtf'), pmRange: getVal('pmRange'), pdRange: getVal('pdRange') }); } // 2. CLOSING EXISTING TRADE: Ignore Ticker/Type, Require Context + Exit else { const idx = trades.findIndex(t => String(t.id) === String(editId)); const ex = parseFloat(document.getElementById('exit').value); const en = trades[idx].entry; const req = ['setup', 'vwap', 'ema512', 'mtf', 'pmRange', 'pdRange']; let missing = false; req.forEach(k => { if(!getVal(k)) { document.querySelector(`[data-id="${k}"]`).classList.add('error-glow'); missing = true; } }); if(missing || isNaN(ex)) { alert("STRICT CLOSE: Mandatory Context and Exit Price Required!"); return; } trades[idx].status = 'CLOSED'; trades[idx].exit = ex; trades[idx].exitTime = getNYTime(); trades[idx].pnl = ((ex - en) * 100 * trades[idx].count).toFixed(2); trades[idx].pnlPct = (((ex - en) / en) * 100).toFixed(2); req.forEach(k => trades[idx][k] = getVal(k)); } document.querySelectorAll('.temp-ticker').forEach(el => el.remove()); resetForm(); render(); };
